<script>
(() => {
  // --- Config (Nokia-like minimalism) ---
  const CELL = 20;               // pixel size of one grid cell
  const COLS = 30;               // 600 / 20
  const ROWS = 30;               // 600 / 20
  const TICK_MS = 100;           // speed (lower = faster). ~10 FPS feels right.
  const WALLS_KILL = true;       // classic: hitting boundary ends game

  const canvas = document.getElementById('game');
  const ctx = canvas.getContext('2d');
  const scoreEl = document.getElementById('score');
  const overlay = document.getElementById('overlay');
  const finalScoreEl = document.getElementById('finalScore');
  const restartBtn = document.getElementById('restartBtn');

  // Handle high-DPI crispness
  const dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
  canvas.width = canvas.width * dpr;
  canvas.height = canvas.height * dpr;
  canvas.style.width = (canvas.width / dpr) + 'px';
  canvas.style.height = (canvas.height / dpr) + 'px';
  ctx.scale(dpr, dpr);

  // Game state
  let snake, dir, nextDir, food, score, alive, accumulator, lastTs;

  function reset() {
    snake = [
      {x: Math.floor(COLS/2), y: Math.floor(ROWS/2)},
      {x: Math.floor(COLS/2)-1, y: Math.floor(ROWS/2)},
      {x: Math.floor(COLS/2)-2, y: Math.floor(ROWS/2)}
    ];
    dir = {x: 1, y: 0};
    nextDir = {...dir};
    score = 0;
    food = spawnFood();
    alive = true;
    accumulator = 0;
    lastTs = performance.now();
    scoreEl.textContent = '0';
    overlay.hidden = true;
    draw(); // draw initial frame
  }

  function spawnFood() {
    let pos;
    do {
      pos = {
        x: Math.floor(Math.random()*COLS),
        y: Math.floor(Math.random()*ROWS),
      };
    } while (snake && snake.some(s => s.x === pos.x && s.y === pos.y));
    return pos;
  }

  function step() {
    // Commit the queued direction (prevents instant reverse)
    dir = nextDir;

    const head = snake[0];
    const newHead = {x: head.x + dir.x, y: head.y + dir.y};

    // Walls?
    if (WALLS_KILL) {
      if (newHead.x < 0 || newHead.x >= COLS || newHead.y < 0 || newHead.y >= ROWS) {
        return die();
      }
    } else {
      // Wrap (not used here, but easy toggle)
      if (newHead.x < 0) newHead.x = COLS - 1;
      if (newHead.x >= COLS) newHead.x = 0;
      if (newHead.y < 0) newHead.y = ROWS - 1;
      if (newHead.y >= ROWS) newHead.y = 0;
    }

    // Self-collision (check against full current snake)
    if (snake.some(seg => seg.x === newHead.x && seg.y === newHead.y)) {
      return die();
    }

    // Move
    snake.unshift(newHead);

    // Eat?
    if (newHead.x === food.x && newHead.y === food.y) {
      score++;
      scoreEl.textContent = score;
      food = spawnFood();
    } else {
      snake.pop(); // keep length if not eating
    }
  }

  function die() {
    alive = false;
    finalScoreEl.textContent = score;
    document.getElementById('overlayTitle').textContent = 'Game Over';
    overlay.hidden = false;
  }

  // Rendering
  function drawGrid() {
    // ultra subtle grid to mimic old LCD feel
    ctx.strokeStyle = '#0a0a0a';
    ctx.lineWidth = 1;
    ctx.beginPath();
    for (let x=0; x<=COLS; x++) {
      ctx.moveTo(x*CELL + .5, 0);
      ctx.lineTo(x*CELL + .5, ROWS*CELL);
    }
    for (let y=0; y<=ROWS; y++) {
      ctx.moveTo(0, y*CELL + .5);
      ctx.lineTo(COLS*CELL, y*CELL + .5);
    }
    ctx.stroke();
  }

  function draw() {
    // Clear (black background)
    ctx.fillStyle = '#000';
    ctx.fillRect(0,0, COLS*CELL, ROWS*CELL);

    drawGrid();

    // Food (red)
    ctx.fillStyle = getCssVar('--food');
    ctx.fillRect(food.x*CELL, food.y*CELL, CELL, CELL);

    // Snake (green)
    ctx.fillStyle = getCssVar('--snake');
    // draw head slightly rounded for subtle detail
    for (let i=0; i<snake.length; i++) {
      const s = snake[i];
      const r = i === 0 ? 4 : 2; // head vs body
      roundRect(ctx, s.x*CELL, s.y*CELL, CELL, CELL, r, true, false);
    }
  }

  // Main loop with fixed-step via accumulator
  function loop(ts) {
    if (!alive) return; // freeze when dead; restart re-enters loop
    const dt = ts - lastTs;
    lastTs = ts;
    accumulator += dt;

    while (accumulator >= TICK_MS) {
      step();
      accumulator -= TICK_MS;
      if (!alive) break;
    }

    draw();
    requestAnimationFrame(loop);
  }

  // Direction controls
  function queueDir(nx, ny) {
    // Prevent reversing directly into yourself (e.g., going left then right)
    if (nx === -dir.x && ny === -dir.y) return;
    // Also prevent multiple changes within same tick to opposite
    if (nx === -nextDir.x && ny === -nextDir.y) return;
    nextDir = {x: nx, y: ny};
  }

  window.addEventListener('keydown', (e) => {
    switch (e.key) {
      case 'ArrowUp':    e.preventDefault(); queueDir(0,-1); break;
      case 'ArrowDown':  e.preventDefault(); queueDir(0, 1); break;
      case 'ArrowLeft':  e.preventDefault(); queueDir(-1,0); break;
      case 'ArrowRight': e.preventDefault(); queueDir(1, 0); break;
      case 'r': case 'R': if (!alive) start(); break;
    }
  });

  // Touch D-pad
  document.querySelectorAll('.dkey').forEach(btn => {
    btn.addEventListener('click', () => {
      const d = btn.getAttribute('data-dir');
      if (d === 'up') queueDir(0,-1);
      if (d === 'down') queueDir(0, 1);
      if (d === 'left') queueDir(-1,0);
      if (d === 'right') queueDir(1, 0);
    });
  });

  restartBtn.addEventListener('click', () => start());

  function start() {
    reset();
    requestAnimationFrame((t) => {
      lastTs = t;
      requestAnimationFrame(loop);
    });
  }

  // Utilities
  function getCssVar(name) {
    return getComputedStyle(document.documentElement).getPropertyValue(name).trim();
  }

  function roundRect(ctx, x, y, w, h, r, fill = true, stroke = false) {
    if (r < 0) r = 0;
    const rr = Math.min(r, w/2, h/2);
    ctx.beginPath();
    ctx.moveTo(x+rr, y);
    ctx.arcTo(x+w, y,   x+w, y+h, rr);
    ctx.arcTo(x+w, y+h, x,   y+h, rr);
    ctx.arcTo(x,   y+h, x,   y,   rr);
    ctx.arcTo(x,   y,   x+w, y,   rr);
    ctx.closePath();
    if (fill) ctx.fill();
    if (stroke) ctx.stroke();
  }

  // Boot
  start();
})();
</script>
</body>
